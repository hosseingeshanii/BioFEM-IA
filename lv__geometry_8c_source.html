54/* 0-based flat index of ring node (k, j):
55 * ring k=0 is the top (near-apex) open boundary ring.
56 * No apex node — both top and base are open boundaries so every
57 * interior vertex has valence 6 (regular subdivision surface). */
58#define RING_NODE(k, j, Np) ((k)*(Np) + (j))
59
60/* 0-based flat index of apex-cap center node i (0,1,2), stored right after
61 * all ring nodes: index N_total*N_phi + i. */
62#define CAP_NODE(i, N_total, Np) ((N_total)*(Np) + (i))

91PetscErrorCode CreateLVMesh(IBMNodes *ibm, FE *fem, const LVParams *p)
92{
93 PetscReal a = p->a;
94 PetscReal b = p->b;
95 PetscReal f_cut = p->f_cut;
96 PetscInt N_theta = p->N_theta;
97 PetscInt N_phi = p->N_phi;
98 PetscInt N_apex_extra = p->N_apex_extra;
99 PetscReal alpha_endo_deg = p->alpha_endo;
100 PetscReal alpha_epi_deg = p->alpha_epi;
101
102 /* Total rings = base mesh rings + extra apex rings */
103 PetscInt N_total = N_theta + N_apex_extra;
104
105 /* Apex cap: 3 interior nodes forming 1 central triangle, fanned out to the
106 * ring-0 boundary (nearest-node assignment split into 3 angular zones).
107 * This trades the old single-node fan's severe singularity (valence
108 * N_phi) for 3 nodes at a more moderate valence (~N_phi/3 + 3), while
109 * most ring-0 nodes only gain +1 valence (one seam node per zone gains
110 * +2) — see design discussion, exact valence 6 throughout is not
111 * reachable with a single-layer 3-node cap when N_phi is large. */
112 const PetscInt N_cap_nodes = 3;
113
114 PetscErrorCode ierr;
115
116 /* ----------------------------------------------------------------
117 * 1. Mesh dimensions
118 * ---------------------------------------------------------------- */
119 /* theta_cut: polar angle where the base plane cuts the spheroid.
120 * z_base = a*cos(theta_cut), theta_cut = acos(1 - 2*f_cut)
121 * Ring k sits at theta = (k+1)*theta_step so ring 0 is one step below
122 * the true apex. Both top and base are open boundaries so every
123 * interior vertex has valence 6 (all regular for Loop subdivision). */
124 PetscReal theta_cut = acos(1.0 - 2.0 * f_cut);
125 PetscReal theta_step = theta_cut / (PetscReal)N_theta;
126 /* PetscReal z_apex = a; */ /* unused: apex cap removed */
127 PetscReal z_base = a * cos(theta_cut);
128
129 /* Cap element count: N_phi-3 zone-interior fan triangles (one per ring-0
130 * edge strictly inside a zone) + 6 seam-bridging triangles (2 per each of
131 * the 3 zone boundaries) + 1 central triangle = N_phi + 4. Every ring-0
132 * edge ends up covered by exactly one new triangle, closing the hole. */
133 PetscInt n_cap_elmt = N_phi + 4;
134
135 PetscInt n_elmt_base = (N_total - 1) * 2 * N_phi; /* quad strips only */
136 PetscInt n_v = N_total * N_phi + N_cap_nodes;
137 /* n_elmt_base intentionally EXCLUDES the cap: IrrVer()/Patch() already
138 * skip ec >= n_elmt_base for Loop-subdivision valence classification and
139 * fall back to CST membrane elements there (see bending.c / active_strain.c) —
140 * exactly the mechanism needed for the cap's irregular valence. */
141 PetscInt n_elmt = n_elmt_base + n_cap_elmt;
142 PetscInt n_edge = 0; /* base BC via EdgeDirectionalFix, no ghosts */
143 PetscInt n_ghosts = 0;
144 /* Apex ring, base ring, AND the 3 new fixed apex-cap nodes are boundary groups. */
145 PetscInt sum_n_bnodes = 2 * N_phi + N_cap_nodes;
146
147 ibm->n_v = n_v;
148 ibm->n_elmt = n_elmt;
149 ibm->n_elmt_base = n_elmt_base;
150 ibm->n_edge = n_edge;
151 ibm->n_ghosts = n_ghosts;
152 ibm->sum_n_bnodes = sum_n_bnodes;
153 ibm->ibi = 0;
154
155 PetscPrintf(PETSC_COMM_WORLD,
156 "LV mesh: a=%.4f b=%.4f f_cut=%.2f theta_cut=%.2f deg "
157 "N_theta=%d N_apex_extra=%d N_phi=%d n_v=%d n_elmt=%d\n",
158 a, b, f_cut, theta_cut * 180.0 / PETSC_PI,
159 (int)N_theta, (int)N_apex_extra, (int)N_phi, (int)n_v, (int)n_elmt);
160
161 /* ----------------------------------------------------------------
162 * 2. Allocate all standard IBMNodes / FE arrays
163 * ---------------------------------------------------------------- */
164 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 2: calling Create()\n");
165 ierr = Create(ibm, fem, 0); CHKERRQ(ierr);
166 PetscPrintf(PETSC_COMM_WORLD, "[lv] Create() done\n");
167
168 ierr = PetscMalloc(sum_n_bnodes * sizeof(PetscInt), &(ibm->bnodes)); CHKERRQ(ierr);
169
170 for (PetscInt i = 0; i < n_elmt; i++) {
171 ibm->El[0][i] = E;
172 ibm->El[1][i] = mu;
173 ibm->E_epsilon[0][i] = 0.0;
174 ibm->E_epsilon[1][i] = 0.0;
175 }
176
177 /* EdgeDirectionalFix unconditionally reads n_bnodes[0..3]; zero-init 8 slots */
178 PetscFree(ibm->n_bnodes);
179 ierr = PetscCalloc1(8, &ibm->n_bnodes); CHKERRQ(ierr);
180 /* group 0: apex ring — fixed via EdgeDirectionalFix(0,...) in main.c, and also
181 * tells IrrVer the apex ring is a boundary (bcount=count+3=6 → regular).
182 * group 1: base ring — currently NOT fixed (main.c only calls edge_n=0 and 2).
183 * group 2: the 3 new apex-cap center nodes — fixed via EdgeDirectionalFix(2,...). */
184 ibm->n_bnodes[0] = N_phi;
185 ibm->n_bnodes[1] = N_phi;
186 ibm->n_bnodes[2] = N_cap_nodes;
187
188 /* ----------------------------------------------------------------
189 * 3. Node coordinates — RING_NODE(k,j,N_phi) = k*N_phi + j
190 * Ring k=0 is the first ring below the apex (θ = theta_step).
191 * ---------------------------------------------------------------- */
192 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 3: filling node coordinates\n");
193
194 /* Two-segment theta spacing:
195 * rings 0..N_apex_extra-1 : squeezed into [0, theta_step] to shrink the apex hole
196 * rings N_apex_extra..N_total-1 : regular spacing theta_step..theta_cut */
197 for (PetscInt k = 0; k < N_total; k++) {
198 PetscReal theta;
199 if (k < N_apex_extra)
200 theta = (PetscReal)(k + 1) / (PetscReal)(N_apex_extra + 1) * theta_step;
201 else
202 theta = (PetscReal)(k - N_apex_extra + 1) * theta_step;
203 PetscReal sth = sin(theta);
204 PetscReal cth = cos(theta);
205 for (PetscInt j = 0; j < N_phi; j++) {
206 PetscReal phi = 2.0 * PETSC_PI * (PetscReal)j / (PetscReal)N_phi;
207 PetscInt nc = RING_NODE(k, j, N_phi);
208 ibm->x_bp[nc] = b * sth * cos(phi);
209 ibm->y_bp[nc] = b * sth * sin(phi);
210 ibm->z_bp[nc] = a * cth;
211 ibm->x_bp0[nc] = ibm->x_bp[nc];
212 ibm->y_bp0[nc] = ibm->y_bp[nc];
213 ibm->z_bp0[nc] = ibm->z_bp[nc];
214 }
215 }
216
217 /* Apex-cap center nodes: 3 nodes on the analytic spheroid surface,
218 * halfway (in theta) between the true apex and ring 0 — using the same
219 * two-segment formula as above so this stays correct whether or not
220 * N_apex_extra squeezes ring 0 closer to the tip — evenly spaced in phi
221 * so they sit at the center of their angular zone (zone i spans phi
222 * roughly [i*120°, (i+1)*120°], see stage 4). */
223 {
224 PetscReal theta_ring0 = (N_apex_extra > 0)
225 ? theta_step / (PetscReal)(N_apex_extra + 1)
226 : theta_step;
227 PetscReal theta_c = 0.5 * theta_ring0;
228 PetscReal sth_c = sin(theta_c);
229 PetscReal cth_c = cos(theta_c);
230 for (PetscInt i = 0; i < N_cap_nodes; i++) {
231 PetscReal phi_c = 2.0 * PETSC_PI * ((PetscReal)i + 0.5) / (PetscReal)N_cap_nodes;
232 PetscInt nc = CAP_NODE(i, N_total, N_phi);
233 ibm->x_bp[nc] = b * sth_c * cos(phi_c);
234 ibm->y_bp[nc] = b * sth_c * sin(phi_c);
235 ibm->z_bp[nc] = a * cth_c;
236 ibm->x_bp0[nc] = ibm->x_bp[nc];
237 ibm->y_bp0[nc] = ibm->y_bp[nc];
238 ibm->z_bp0[nc] = ibm->z_bp[nc];
239 }
240 }
241
242 /* ----------------------------------------------------------------
243 * 4. Element connectivity (0-based node indices)
244 *
245 * Quad strips (base mesh) followed by the 3-node apex cap.
246 * ---------------------------------------------------------------- */
247 PetscInt ec = 0;
248
249 for (PetscInt k = 0; k < N_total - 1; k++) {
250 for (PetscInt j = 0; j < N_phi; j++) {
251 PetscInt jn = (j + 1) % N_phi;
252 PetscInt n_bl = RING_NODE(k, j, N_phi);
253 PetscInt n_br = RING_NODE(k, jn, N_phi);
254 PetscInt n_tl = RING_NODE(k + 1, j, N_phi);
255 PetscInt n_tr = RING_NODE(k + 1, jn, N_phi);
256
257 ibm->nv1[ec] = n_bl; ibm->nv2[ec] = n_tl; ibm->nv3[ec] = n_br; ec++;
258 ibm->nv1[ec] = n_tl; ibm->nv2[ec] = n_tr; ibm->nv3[ec] = n_br; ec++;
259 }
260 }
261
262 /* n_elmt_base excludes the cap by design (see comment at its definition
263 * above) — ec must equal n_elmt_base exactly at this point. */
264 if (ec != n_elmt_base) {
265 SETERRQ2(PETSC_COMM_WORLD, PETSC_ERR_PLIB,
266 "LV mesh: base element count mismatch (ec=%d, n_elmt_base=%d)",
267 (int)ec, (int)n_elmt_base);
268 }
269
270 /* ----------------------------------------------------------------
271 * 4b. Apex cap — 3 center nodes fanned out to ring 0.
272 *
273 * Ring-0 nodes are split into 3 contiguous angular zones (nearest-
274 * center assignment). Each zone is fanned to its center node; the 3
275 * zone boundaries ("seams") are bridged by 2 triangles each, reusing
276 * the center triangle's edges so every ring-0 edge ends up covered by
277 * exactly one new triangle and the mesh stays manifold.
278 * ---------------------------------------------------------------- */
279 {
280 PetscInt zone_bound[4];
281 for (PetscInt i = 0; i <= N_cap_nodes; i++) {
282 zone_bound[i] = (PetscInt)round((PetscReal)i * (PetscReal)N_phi
283 / (PetscReal)N_cap_nodes);
284 }
285
286 for (PetscInt i = 0; i < N_cap_nodes; i++) {
287 PetscInt c_i = CAP_NODE(i, N_total, N_phi);
288 /* zone-interior fan: one triangle per ring-0 edge strictly inside the zone */
289 for (PetscInt k = zone_bound[i]; k < zone_bound[i + 1] - 1; k++) {
290 ibm->nv1[ec] = RING_NODE(0, k, N_phi);
291 ibm->nv2[ec] = RING_NODE(0, k + 1, N_phi);
292 ibm->nv3[ec] = c_i;
293 ec++;
294 }
295 }
296
297 for (PetscInt i = 0; i < N_cap_nodes; i++) {
298 PetscInt i_next = (i + 1) % N_cap_nodes;
299 PetscInt c_i = CAP_NODE(i, N_total, N_phi);
300 PetscInt c_next = CAP_NODE(i_next, N_total, N_phi);
301 PetscInt r_last = zone_bound[i + 1] - 1; /* last node of zone i */
302 PetscInt r_seam = zone_bound[i + 1] % N_phi; /* first node of zone i+1 */
303
304 /* seam bridge A: covers the ring-0 boundary edge (r_last, r_seam) */
305 ibm->nv1[ec] = RING_NODE(0, r_last, N_phi);
306 ibm->nv2[ec] = RING_NODE(0, r_seam, N_phi);
307 ibm->nv3[ec] = c_i;
308 ec++;
309
310 /* seam bridge B: fills the gap up to the center triangle's edge */
311 ibm->nv1[ec] = c_next;
312 ibm->nv2[ec] = c_i;
313 ibm->nv3[ec] = RING_NODE(0, r_seam, N_phi);
314 ec++;
315 }
316
317 /* center triangle */
318 ibm->nv1[ec] = CAP_NODE(0, N_total, N_phi);
319 ibm->nv2[ec] = CAP_NODE(1, N_total, N_phi);
320 ibm->nv3[ec] = CAP_NODE(2, N_total, N_phi);
321 ec++;
322
323 /* Orientation fix: for a star-shaped-about-origin surface like this
324 * spheroid, the outward normal always has positive dot product with
325 * the element centroid. Swap nv2/nv3 on any cap triangle that fails
326 * this test so all cap elements share consistent outward winding with
327 * the rest of the mesh, regardless of the manual ordering above. */
328 for (PetscInt e = n_elmt_base; e < ec; e++) {
329 PetscInt n1 = ibm->nv1[e], n2 = ibm->nv2[e], n3 = ibm->nv3[e];
330 struct Cmpnts p1 = {ibm->x_bp[n1], ibm->y_bp[n1], ibm->z_bp[n1]};
331 struct Cmpnts p2 = {ibm->x_bp[n2], ibm->y_bp[n2], ibm->z_bp[n2]};
332 struct Cmpnts p3 = {ibm->x_bp[n3], ibm->y_bp[n3], ibm->z_bp[n3]};
333 struct Cmpnts centroid = AMULT(1.0 / 3.0, PLUS(PLUS(p1, p2), p3));
334 struct Cmpnts normal = CROSS(MINUS(p2, p1), MINUS(p3, p1));
335 if (DOT(normal, centroid) < 0.0) {
336 ibm->nv2[e] = n3;
337 ibm->nv3[e] = n2;
338 }
339 }
340 }
341
342 if (ec != n_elmt) {
343 SETERRQ2(PETSC_COMM_WORLD, PETSC_ERR_PLIB,
344 "LV mesh: total element count mismatch (ec=%d, n_elmt=%d)",
345 (int)ec, (int)n_elmt);
346 }
347
348 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 4: connectivity done (%d elements)\n", (int)ec);
349
350 /* ----------------------------------------------------------------
351 * 5. Patch nodes nv4/nv5/nv6
352 * Opposite node of each neighbouring triangle — 1000000 = none.
353 * ---------------------------------------------------------------- */
354 for (PetscInt i = 0; i < n_elmt; i++) {
355 ibm->nv4[i] = 1000000;
356 ibm->nv5[i] = 1000000;
357 ibm->nv6[i] = 1000000;
358 }
359
360 for (PetscInt i = 0; i < n_elmt; i++) {
361 PetscInt n1e = ibm->nv1[i], n2e = ibm->nv2[i], n3e = ibm->nv3[i];
362 for (PetscInt j = 0; j < n_elmt; j++) {
363 if (i == j) continue;
364 PetscInt n1pe = ibm->nv1[j], n2pe = ibm->nv2[j], n3pe = ibm->nv3[j];
365 PetscInt mn = 0;
366 PetscReal cn = 0.0;
367 if (n1e == n1pe || n1e == n2pe || n1e == n3pe) { mn++; cn += 3.5; }
368 if (n2e == n1pe || n2e == n2pe || n2e == n3pe) { mn++; cn += 2.5; }
369 if (n3e == n1pe || n3e == n2pe || n3e == n3pe) { mn++; cn += 1.5; }
370 if (mn == 2) {
371 PetscInt npe;
372 if (n1pe != n1e && n1pe != n2e && n1pe != n3e) npe = n1pe;
373 else if (n2pe != n1e && n2pe != n2e && n2pe != n3e) npe = n2pe;
374 else npe = n3pe;
375 if (cn == 4.0) ibm->nv4[i] = npe;
376 else if (cn == 5.0) ibm->nv5[i] = npe;
377 else ibm->nv6[i] = npe;
378 }
379 }
380 }
381
382 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 5: patch nodes done\n");
383
384 /* ----------------------------------------------------------------
385 * 6. Boundary rings — top (ring 0) and base (ring N_theta-1)
386 * ---------------------------------------------------------------- */
387 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 6: boundary edge info\n");
388 /* Both apex ring (edge_n=0) and base ring (edge_n=1) in bnodes[].
389 IrrVer uses this to apply the ghost trick (bcount=count+3) for boundary nodes,
390 keeping the apex ring classified as REGULAR despite the cap additions.
391 No displacement BCs are applied to the apex ring — EdgeDirectionalFix targets edge_n=1 only. */
392 for (PetscInt j = 0; j < N_phi; j++)
393 ibm->bnodes[j] = RING_NODE(0, j, N_phi); /* apex ring (edge_n=0) */
394 for (PetscInt j = 0; j < N_phi; j++)
395 ibm->bnodes[N_phi + j] = RING_NODE(N_total - 1, j, N_phi); /* base ring (edge_n=1) */
396
397 /* ----------------------------------------------------------------
398 * 7. Fiber directions — Streeter helical rule
399 *
400 * At each element centroid c = (cx, cy, cz):
401 *
402 * e_n = UNIT( cx/b², cy/b², cz/a² ) outward spheroid normal
403 * e_l = UNIT( ẑ₋ - (ẑ₋·e_n) e_n ) meridional apex→base
404 * where ẑ₋ = (0,0,-1)
405 * e_c = e_n × e_l circumferential
406 *
407 * α = (α_endo + α_epi)/2 mid-wall value, constant (Bayer Eq.2, d=0.5)
408 *
409 * f = cos(α)·e_c + sin(α)·e_l → ibm->n_fib[ec]
410 * ---------------------------------------------------------------- */
411 PetscPrintf(PETSC_COMM_WORLD, "[lv] stage 7: assigning fiber directions\n");
412
413 /* Mid-wall helix angle: Bayer 2012 Eq.(2) evaluated at d=0.5 */
414 PetscReal alpha = 0.5 * (alpha_endo_deg + alpha_epi_deg) * PETSC_PI / 180.0;
415
416 /* Direction vector pointing from apex toward base along the z axis */
417 struct Cmpnts e_down = {0.0, 0.0, -1.0};
418
419 for (ec = 0; ec < n_elmt; ec++) {
420 PetscInt n1e = ibm->nv1[ec], n2e = ibm->nv2[ec], n3e = ibm->nv3[ec];
421
422 /* Element centroid */
423 struct Cmpnts c;
424 c.x = (ibm->x_bp[n1e] + ibm->x_bp[n2e] + ibm->x_bp[n3e]) / 3.0;
425 c.y = (ibm->y_bp[n1e] + ibm->y_bp[n2e] + ibm->y_bp[n3e]) / 3.0;
426 c.z = (ibm->z_bp[n1e] + ibm->z_bp[n2e] + ibm->z_bp[n3e]) / 3.0;
427
428 /* Outward surface normal: gradient of the spheroid level-set F=x²/b²+y²/b²+z²/a²-1 */
429 struct Cmpnts grad_F;
430 grad_F.x = c.x / (b * b);
431 grad_F.y = c.y / (b * b);
432 grad_F.z = c.z / (a * a);
433 struct Cmpnts e_n = UNIT(grad_F);
434
435 /* Meridional direction (apex→base): project e_down onto the tangent plane */
436 PetscReal d = DOT(e_down, e_n);
437 struct Cmpnts e_l_raw;
438 e_l_raw.x = e_down.x - d * e_n.x;
439 e_l_raw.y = e_down.y - d * e_n.y;
440 e_l_raw.z = e_down.z - d * e_n.z;
441
442 struct Cmpnts e_l;
443 if (SIZE(e_l_raw) > 1.0e-10) {
444 e_l = UNIT(e_l_raw);
445 } else {
446 /* Degenerate at or very near the apex: fall back to a stable tangent.
447 * Project (1,0,0) onto the tangent plane. */
448 struct Cmpnts e_x = {1.0, 0.0, 0.0};
449 PetscReal d2 = DOT(e_x, e_n);
450 e_l_raw.x = e_x.x - d2 * e_n.x;
451 e_l_raw.y = e_x.y - d2 * e_n.y;
452 e_l_raw.z = e_x.z - d2 * e_n.z;
453 e_l = UNIT(e_l_raw);
454 }
455
456 /* Circumferential direction (right-hand: outward-normal × meridional) */
457 struct Cmpnts e_c = CROSS(e_n, e_l);
458
459 /* Fiber unit vector in the tangent plane — Bayer 2012 Eq.(7) */
460 ibm->n_fib[ec].x = cos(alpha) * e_c.x + sin(alpha) * e_l.x;
461 ibm->n_fib[ec].y = cos(alpha) * e_c.y + sin(alpha) * e_l.y;
462 ibm->n_fib[ec].z = cos(alpha) * e_c.z + sin(alpha) * e_l.z;
463 }

465 PetscPrintf(PETSC_COMM_WORLD,
