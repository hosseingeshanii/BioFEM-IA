18static PetscReal MaxCmpntsDiff_(Cmpnts a, Cmpnts b)
19{
20 PetscReal max_diff = PetscAbsReal(a.x - b.x);
21 max_diff = PetscMax(max_diff, PetscAbsReal(a.y - b.y));
22 max_diff = PetscMax(max_diff, PetscAbsReal(a.z - b.z));
23 return max_diff;
24}

26static PetscReal MaxMat3Diff_(const PetscReal A[3][3], const PetscReal B[3][3])
27{
28 PetscReal max_diff = 0.0;
29 for (PetscInt i = 0; i < 3; ++i) {
30 for (PetscInt j = 0; j < 3; ++j) {
31 max_diff = PetscMax(max_diff, PetscAbsReal(A[i][j] - B[i][j]));
32 }
33 }
34 return max_diff;

84static PetscErrorCode CompareMetricState_(FE *fem,
85 PetscInt body_id,
86 PetscInt n_compare_elems,
87 const MetricSnapshotEntry *cpu_snapshot)
88{
89 PetscFunctionBeginUser;
90
91 PetscCheck(fem != NULL, PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "FE pointer must not be NULL");
92 PetscCheck(cpu_snapshot != NULL, PETSC_COMM_SELF, PETSC_ERR_ARG_NULL,
93 "cpu_snapshot must not be NULL");
94
95 const PetscInt n_qp = fem->act_data.n_qp;
96 const PetscReal tol = 1.0e-9;
97
98 PetscPrintf(PETSC_COMM_SELF,
99 "\nCPU vs CUDA metric comparison for body %d (first %" PetscInt_FMT " elements)\n",
100 (int)body_id, n_compare_elems);
101
102 for (PetscInt ec = 0; ec < n_compare_elems; ++ec) {
103 PetscReal max_g = 0.0, max_g0 = 0.0, max_gm = 0.0, max_gm0 = 0.0;
104 for (PetscInt qp = 0; qp < n_qp; ++qp) {
105 const size_t idx = (size_t)ec * (size_t)n_qp + (size_t)qp;
106 ElemActData *ead = &fem->act_data.elem_act_data[ec];
107
108 for (PetscInt i = 0; i < 3; ++i) {
109 max_g = PetscMax(max_g, MaxCmpntsDiff_(cpu_snapshot[idx].g.Cov[i], ead->g[qp].Cov[i]));
110 max_g = PetscMax(max_g, MaxCmpntsDiff_(cpu_snapshot[idx].g.Cont[i], ead->g[qp].Cont[i]));
111 max_g0 = PetscMax(max_g0, MaxCmpntsDiff_(cpu_snapshot[idx].g0.Cov[i], ead->g0[qp].Cov[i]));
112 max_g0 = PetscMax(max_g0, MaxCmpntsDiff_(cpu_snapshot[idx].g0.Cont[i], ead->g0[qp].Cont[i]));
113 }
114
115 max_gm = PetscMax(max_gm, MaxMat3Diff_(cpu_snapshot[idx].gm.Cov, ead->gm[qp].Cov));
116 max_gm = PetscMax(max_gm, MaxMat3Diff_(cpu_snapshot[idx].gm.Cont, ead->gm[qp].Cont));
117 max_gm0 = PetscMax(max_gm0, MaxMat3Diff_(cpu_snapshot[idx].gm0.Cov, ead->gm0[qp].Cov));
118 max_gm0 = PetscMax(max_gm0, MaxMat3Diff_(cpu_snapshot[idx].gm0.Cont, ead->gm0[qp].Cont));
119 }
120
121 PetscPrintf(PETSC_COMM_SELF,
122 " elem %" PetscInt_FMT ": max|g|=% .3e max|g0|=% .3e max|gm|=% .3e max|gm0|=% .3e -> %s\n",
123 ec, (double)max_g, (double)max_g0, (double)max_gm, (double)max_gm0,
124 (max_g <= tol && max_g0 <= tol && max_gm <= tol && max_gm0 <= tol) ? "MATCH" : "DIFF");
125 }
126
127 PetscFunctionReturn(0);
128}

201PetscErrorCode RunCudaProcesses(FE *fem)
202{

204
205 PetscFunctionBeginUser;
206
207 PetscCheck(fem != NULL, PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "FE pointer must not be NULL");
208 PetscCheck(muscle_activation, PETSC_COMM_SELF, PETSC_ERR_ARG_WRONG,
209 "-cuda_process requires -muscle_activation 1 so metric tensor data is allocated");
210
211 ierr = InitCudaWorkspace(); CHKERRQ(ierr);
212
213 for (PetscInt ibi = 0; ibi < nbody; ibi++) {
214 const PetscInt n_compare_elems = PetscMin((PetscInt)3, fem[ibi].ibm->n_elmt);
215 MetricSnapshotEntry *cpu_snapshot = NULL;
216 double t0, t1;
217 double cpu_geom0_time, cpu_geom_time, cpu_metric_time;
218 double cuda_geom0_time, cuda_geom_time, cuda_metric_time;

220 t0 = MPI_Wtime();

222 t1 = MPI_Wtime();
223 cpu_geom0_time = t1 - t0;
224
225 t0 = MPI_Wtime();
226 ierr = UpdateElements(&fem[ibi], ElemUpdateGeomSubdiv); CHKERRQ(ierr);
227 t1 = MPI_Wtime();
228 cpu_geom_time = t1 - t0;
229
230 t0 = MPI_Wtime();
231 ierr = UpdateElements(&fem[ibi], ElemUpdateG); CHKERRQ(ierr);
232 t1 = MPI_Wtime();
233 cpu_metric_time = t1 - t0;
234
235 ierr = SnapshotMetricState_(&fem[ibi], n_compare_elems, &cpu_snapshot); CHKERRQ(ierr);
236
237 ierr = PrepareCudaHostWorkspace(&fem[ibi]); CHKERRQ(ierr);
238
239 // Run multiple iterations to measure allocation reuse timing
240 const PetscInt n_subdiv_iterations = 5;
241 double cuda_geom0_times[5] = {0};
242 double cuda_geom_times[5] = {0};
243
244 for (PetscInt iter = 0; iter < n_subdiv_iterations; ++iter) {
245 if (iter == 0) {
246 t0 = MPI_Wtime();
247 ierr = RunCudaSubdivGeomKernel(&fem[ibi], PETSC_TRUE); CHKERRQ(ierr);
248 t1 = MPI_Wtime();
249 cuda_geom0_times[iter] = t1 - t0;
250 }
251
252 t0 = MPI_Wtime();
253 ierr = RunCudaSubdivGeomKernel(&fem[ibi], PETSC_FALSE); CHKERRQ(ierr);
254 t1 = MPI_Wtime();
255 cuda_geom_times[iter] = t1 - t0;
256 }
257
258 cuda_geom0_time = cuda_geom0_times[0];
259
260 double cuda_geom_reuse_total = 0.0;
261 PetscInt n_cuda_geom_reuse = 0;
262 for (PetscInt iter = 1; iter < n_subdiv_iterations; ++iter) {
263 cuda_geom_reuse_total += cuda_geom_times[iter];
264 ++n_cuda_geom_reuse;
265 }
266 cuda_geom_time = (n_cuda_geom_reuse > 0) ? cuda_geom_reuse_total / (double)n_cuda_geom_reuse : cuda_geom_times[0];
267
268 // Print timing comparison for each iteration
269 for (PetscInt iter = 0; iter < n_subdiv_iterations; ++iter) {
270 if (iter == 0) {
271 PetscPrintf(PETSC_COMM_SELF,
272 "[iter %" PetscInt_FMT "] SubdivGeomKernel (reference): CUDA=%.6e CPU=%.6e speedup=%.3f x (includes allocation)\n",
273 iter, cuda_geom0_times[iter], cpu_geom0_time,
274 (double)TimingSpeedup_(cpu_geom0_time, cuda_geom0_times[iter]));
275 }
276
277 PetscPrintf(PETSC_COMM_SELF,
278 "[iter %" PetscInt_FMT "] SubdivGeomKernel (current): CUDA=%.6e CPU=%.6e speedup=%.3f x%s\n",
279 iter, cuda_geom_times[iter], cpu_geom_time,
280 (double)TimingSpeedup_(cpu_geom_time, cuda_geom_times[iter]),
281 (iter == 0) ? " (same outer iter as reference)" : " (reuse)");
282 }
283
284 t0 = MPI_Wtime();
285 ierr = RunCudaMetricTensorKernel(&fem[ibi]); CHKERRQ(ierr);
286 t1 = MPI_Wtime();
287 cuda_metric_time = t1 - t0;
288 PetscPrintf(PETSC_COMM_SELF,
289 "CUDA metric-tensor kernel finished successfully for body %d.\n",
290 ibi);
291
292 ierr = PrintCudaPreprocessTiming_(ibi,
293 cpu_geom0_time,
294 cpu_geom_time,
295 cpu_metric_time,
296 0.0, // cuda_coord_time (removed - was redundant with SubdivGeomKernel)
297 cuda_geom0_time,
298 cuda_geom_time,
299 cuda_metric_time); CHKERRQ(ierr);
300 ierr = CompareMetricState_(&fem[ibi], ibi, n_compare_elems, cpu_snapshot); CHKERRQ(ierr);
301 ierr = PrintMetricSampleValues_(&fem[ibi], ibi, n_compare_elems, cpu_snapshot); CHKERRQ(ierr);
302 ierr = PetscFree(cpu_snapshot); CHKERRQ(ierr);
303 }
304
305 PetscFunctionReturn(0);
