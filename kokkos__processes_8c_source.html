129static PetscReal MaxGeomDiff_(const SubdivGeomQP *cpu, const SubdivGeomQP *kokkos)
130{
131 PetscReal max_diff = 0.0;
132 max_diff = PetscMax(max_diff, IntDiff_(cpu->is_irregular, kokkos->is_irregular));
133 max_diff = PetscMax(max_diff, IntDiff_(cpu->v, kokkos->v));
134 max_diff = PetscMax(max_diff, IntDiff_(cpu->nen, kokkos->nen));
135 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->ndx21, kokkos->ndx21));
136 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->ndx31, kokkos->ndx31));
137 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->nn, kokkos->nn));
138 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->gc1, kokkos->gc1));
139 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->gc2, kokkos->gc2));
140 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->Aaa, kokkos->Aaa));
141 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->Abb, kokkos->Abb));
142 max_diff = PetscMax(max_diff, MaxCmpntsDiff_(cpu->Aab, kokkos->Aab));
143
144 if (cpu->is_irregular && kokkos->is_irregular && cpu->nen == kokkos->nen) {
145 max_diff = PetscMax(max_diff, MaxIrregularArrayDiff_(cpu->INa0, kokkos->INa0, cpu->nen));
146 max_diff = PetscMax(max_diff, MaxIrregularArrayDiff_(cpu->INa1, kokkos->INa1, cpu->nen));
147 max_diff = PetscMax(max_diff, MaxIrregularArrayDiff_(cpu->INab0, kokkos->INab0, cpu->nen));
148 max_diff = PetscMax(max_diff, MaxIrregularArrayDiff_(cpu->INab1, kokkos->INab1, cpu->nen));
149 max_diff = PetscMax(max_diff, MaxIrregularArrayDiff_(cpu->INab2, kokkos->INab2, cpu->nen));
150 }
151
152 return max_diff;
153}
