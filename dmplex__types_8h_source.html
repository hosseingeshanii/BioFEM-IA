16/**
17 * @brief Per-rank local copy of Loop subdivision patch stencil data.
18 *
19 * Built once from the global IBMNodes arrays after DMPlex distribution.
20 * All arrays are indexed by local cell index lc in [0, nLocalCells).
21 * The DM reference is retained so that downstream functions can query it.
22 */
23typedef struct {
24 DM dm; /**< Distributed DM; ref-counted, destroyed with layout. */
25 PetscInt *orig_cell; /**< orig_cell[lc]: original IBMNodes cell index for local cell lc.
26 -1 for ghost cells whose stencil is unavailable. */
27 PetscInt *ire; /**< ire[lc]: Loop subdivision flag (0=regular, 1=irregular). */
28 PetscInt *irv; /**< irv[lc]: irregular-vertex flag. */
29 PetscInt *val; /**< val[lc]: valence of the irregular vertex (used when ire==1). */
30 PetscInt *patch; /**< patch[16*lc+i]: original vertex index of the i-th patch node.
31 Up to 12 entries for regular (ire==0) and val+6 for irregular.
32 Unused slots hold DMPLEX_GEOM_PATCH_MISSING (1000000). */
33 PetscInt cStart; /**< First cell point in the DM chart (height stratum 0). */
34 PetscInt cEnd; /**< One past the last cell point. */
35 PetscInt nLocalCells; /**< Number of locally owned cells: cEnd - cStart. */
36} DMPlexPatchLayout;

38/**
39 * @brief MPI coordinate exchange for cross-rank patch stencil nodes.
40 *
41 * Holds the PetscSF and receive buffers used to fetch vertex coordinates that
42 * belong to other MPI ranks but are referenced by this rank's patch stencils.
43 *
44 * SF layout: nroots = ibm->n_v (full original vertex array; each rank serves as
45 * potential root for all n_v entries), nleaves = nRemoteNodes.
46 *
47 * Data flow each geometry iteration:
48 * @code
49 * PetscSFBcast(sf, ibm->x_bp → x_remote) // fetch from owning ranks
50 * ibm->x_bp[orig_node[i]] = x_remote[i] // write-back
51 * @endcode
52 */
53typedef struct {
54 PetscSF sf; /**< Star Forest: nroots=ibm->n_v, nleaves=nRemoteNodes. */
55 PetscInt nRemoteNodes; /**< Number of unique remote vertices needed by this rank. */
56 PetscInt *orig_node; /**< orig_node[i]: original vertex index of remote node i. */
57 PetscMPIInt *owner_rank; /**< owner_rank[i]: MPI rank that owns orig_node[i]. */
58 PetscInt *owner_index; /**< owner_index[i]: root slot on owner_rank[i] (= orig_node[i]). */
59 PetscReal *x_remote; /**< Receive buffer for x coordinates. Size: nRemoteNodes. */
60 PetscReal *y_remote; /**< Receive buffer for y coordinates. Size: nRemoteNodes. */
61 PetscReal *z_remote; /**< Receive buffer for z coordinates. Size: nRemoteNodes. */
62} DMPlexPatchExchange;

90 -1 before that call. */
91 PetscBool initialized; /**< PETSC_TRUE after FEM_DMPlexGeomSetup completes. */
92 MPI_Comm comm; /**< Communicator used during setup. */
93} DMPlexGeomCtx;
